import os
import warnings
import numpy as np
import pandas as pd
import kagglehub
from kagglehub import KaggleDatasetAdapter
from sklearn.model_selection import train_test_split
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import OneHotEncoder, StandardScaler
from sklearn.impute import SimpleImputer

warnings.filterwarnings("ignore")
os.environ["TQD_OMIT_SIZE_DIV"] = "1"

df = kagglehub.dataset_load(
    KaggleDatasetAdapter.PANDAS,
    "redwankarimsony/heart-disease-data",
    "heart_disease_uci.csv"
)

df["num"] = (pd.to_numeric(df["num"], errors="coerce").fillna(0) > 0).astype(int)
dropped_system = ["id", "dataset"]
df = df.drop(columns=dropped_system, errors="ignore")
print(f"  : {dropped_system}")

cols_fix = ["trestbps", "chol", "thalch", "oldpeak"]
print(f"   : {cols_fix}")
for col in cols_fix:
    if col in df.columns:
        invalid_mask = (df[col] <= 0) if col != "oldpeak" else (df[col] < 0)
        count = invalid_mask.sum()
        if count > 0:
            df.loc[invalid_mask, col] = np.nan
            print(f"    -   '{col}'     NaN - {count}  ")

THRESHOLD = 0.4
missing_frac = df.isna().mean()
cols_to_drop = missing_frac[missing_frac > THRESHOLD].index.tolist()
cols_to_drop = [c for c in cols_to_drop if c != "num"]
df = df.drop(columns=cols_to_drop, errors="ignore")
print(f"    (> {THRESHOLD*100}%): {cols_to_drop}")

X = df.drop(columns=["num"])
y = df["num"]
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.20, random_state=42, stratify=y
)

num_cols = X_train.select_dtypes(include=["number"]).columns.tolist()
cat_cols = X_train.select_dtypes(exclude=["number"]).columns.tolist()
print(f"   (StandardScale + Median Impute): {num_cols}")
print(f"   (One-Hot + Freq Impute): {cat_cols}")

preprocessor = ColumnTransformer(
    transformers=[
        ("num", Pipeline([
            ("imputer", SimpleImputer(strategy="median")),
            ("scaler", StandardScaler())
        ]), num_cols),
        ("cat", Pipeline([
            ("imputer", SimpleImputer(strategy="most_frequent")),
            ("onehot", OneHotEncoder(handle_unknown="ignore", sparse_output=False, drop="if_binary"))
        ]), cat_cols)
    ],
    verbose_feature_names_out=False
)

X_train_processed = preprocessor.fit_transform(X_train)
X_test_processed = preprocessor.transform(X_test)
feature_names = preprocessor.get_feature_names_out().tolist()

new_features = [f for f in feature_names if any(cat in f for cat in cat_cols)]
print(f"   : {new_features}")
print(f"  : {X_train_processed.shape}")
